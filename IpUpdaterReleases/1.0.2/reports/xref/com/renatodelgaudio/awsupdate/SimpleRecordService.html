44  	private final static Logger log = LoggerFactory.getLogger(SimpleRecordService.class);
45  	@Autowired
46  	Configuration config;

48  	@Override
49  	public String getCurrentIP() {
50  		ResourceRecordSet set = getCurrentRecordSet();
51  		if (set!=null && set.getResourceRecords() != null && set.getResourceRecords().size() > 0){
52  			ResourceRecord	rr = set.getResourceRecords().get(0);
53  			if(rr!=null){
54  				return rr.getValue();
55  			}
56  		}
57  		return null;
58  	}
59  	
60  	public ResourceRecordSet getCurrentRecordSet() {
61  		
62  		if (log.isDebugEnabled()){
63  			log.debug("zoneId:"+config.getZoneId());
64  			log.debug("recordName:"+config.getRecordName());
65  		}
66  		ListResourceRecordSetsRequest request = new ListResourceRecordSetsRequest();
67  		request.setHostedZoneId(config.getZoneId());
68  
69  		ListResourceRecordSetsResult result = config.getAmazonClient().listResourceRecordSets(request);
70  		List<ResourceRecordSet> recordSets = result.getResourceRecordSets();
71  
72  
73  		for(ResourceRecordSet recordSet : recordSets) {
74  
75  			String currName = recordSet.getName();
76  
77  			if (currName == null)
78  				continue;
79  			currName = currName.trim();
80  
81  			// AWS ends name with dot.
82  			if (currName.endsWith("."))
83  				currName = currName.substring(0,currName.length()-1);
84  
85  			if (equalsIgnoreCase(config.getRecordName(),currName)) {			
86  				return recordSet;
87  			}
88  		}
89  
90  		return null;
91  	}

93  	@Override
94  	public boolean updateRecord(String ip) {
95  	    
96  	    String recordName = config.getRecordName();
97  		
98  		ChangeResourceRecordSetsRequest changeRequest = new ChangeResourceRecordSetsRequest();
99  		changeRequest.setHostedZoneId(config.getZoneId());
100 		ChangeBatch batch = new ChangeBatch();
101 		Change change = new Change();
102 		
103 		
104 		ResourceRecordSet set = getCurrentRecordSet();
105 		if (set!=null){
106 			if (!equalsIgnoreCase("A", set.getType())){
107 				log.error("Record already exists but not as Type A. No actions were performed.");
108 				return false;
109 			}
110 			change.setAction("UPSERT");
111 			log.info("Record ["+recordName+"] already present on AWS Route 53. Upating it..");
112 		} else{
113 			change.setAction("CREATE");
114 			log.info("Record ["+recordName+"] not present on AWS Route 53. Creating it..");
115 			set = new ResourceRecordSet().withName(recordName).withType("A");
116 		}
117 
118 		
119 		set.setTTL(Long.parseLong(config.getTTL()));
120 		List<ResourceRecord> l = new ArrayList<ResourceRecord>();
121 		l.add(new ResourceRecord(ip));
122 		set.setResourceRecords(l);
123 
124 		change.setResourceRecordSet(set);
125 		batch.withChanges(change);
126 
127 		changeRequest.setChangeBatch(batch);
128 
129 		log.info("Updating DNS "+recordName+" with IP "+ip);
130 	
131 		ChangeResourceRecordSetsResult result =  config.getAmazonClient().changeResourceRecordSets(changeRequest);
132 		log.info(result.toString()); 
133 		return true;
134 	}
