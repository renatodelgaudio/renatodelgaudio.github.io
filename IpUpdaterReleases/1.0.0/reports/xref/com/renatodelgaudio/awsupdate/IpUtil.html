32  public class IpUtil {

37  	/**
38  	 * 
39  	 * @return the known IP Provider in random order
40  	 */
41  	public static List<IpProvider> getIpProviders(){
42  
43  		IpProvider providers[] = new IpProvider[2];
44  		providers[0] = new IcanHazip();
45  		providers[1] = new IpInfo();
46  
47  		int mainIndex = (int) Math.round(Math.random());
48  		int fallBackIndex = ~mainIndex + 2;
49  
50  		List<IpProvider> ret = new ArrayList<IpProvider>();
51  		ret.add(providers[mainIndex]);
52  		ret.add(providers[fallBackIndex]);
53  		return ret;
54  
55  	}
56  
57  	/**
58  	 * It retrieves the Public IP trying multiple providers (if needed)
59  	 * @return
60  	 * @throws IpRetrievalException if no IP can be retrieved from the available providers or an invalid IP is returned
61  	 */
62  	public static String retrievePublicIP() throws IpRetrievalException{
63  
64  		List<IpProvider> providers = getIpProviders();
65  		String ip = null;
66  		IpProvider provider = null;
67  		for(IpProvider p : providers){
68  			try{
69  				provider = p;
70  				ip = p.getIP();
71  				break;
72  			}catch(IpRetrievalException e){
73  				log.error("Could not retrieve the public IP from "+provider.providerName(),e);
74  				continue;
75  			}
76  		}
77  
78  		if (!new IPAddressValidator().validate(ip)){
79  			log.error(ip+ " is not a valid IP. Exiting...");
80  			IpRetrievalException ex = new IpRetrievalException(ip+ " is not a valid IP. Exiting...");
81  			ex.setProviderName(provider.providerName());
82  			throw ex;
83  		}
84  		// validate IP
85  		if (isBlank(ip)){
86  			log.error("Cannot get public IP. Exiting...");
87  			IpRetrievalException ex = new IpRetrievalException("Cannot get public IP. Exiting...");
88  			ex.setProviderName(provider.providerName());
89  			throw ex;
90  		}
91  
92  		return ip;		
93  	}
